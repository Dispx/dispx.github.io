Binary tree

This class contains method that takes numbers as input and creates binary tree structure and prints the tree out in console

Code snippet

class BinaryTree
    {
        public Node root { get; set; }

        //Checks if root node is empty. If root node is not empty method calls the PrintHelper method.
        public void Print()
        {
            if (root != null)
            {
                PrintHelper("-", root);
            }
            else
            {
                Console.WriteLine("Root is empty.");
            }
        }

        // Recursively prints the binary tree.
        public void PrintHelper(string lvl, Node n)
        {            
            Console.WriteLine(lvl+" :" + n.number);
            lvl += "-";
            if (n.left != null)
            {                
                PrintHelper(lvl,n.left);
            }
            if (n.right != null)
            {                
                PrintHelper(lvl, n.right);
            }
        }

        //Creates new node with input data and loops through the binary tree to insert node in appropriate position.
        public void Add(int number)
        {
            Node n = new Node(number);
            if (root == null)
            {
                root = n;
            }
            else
            {
                Node i = root;
                while (true)
                {
                    if (i.number > n.number)
                    {
                        if (i.left == null)
                        {
                            i.left = n;
                            break;
                        }
                        else
                        {
                            i = i.left;
                        }
                    }
                    else
                    {
                        if (i.right == null)
                        {
                            i.right = n;
                            break;
                        }
                        else
                        {
                            i = i.right;
                        }
                    }
                }
            }
        }        

        //Loops through the binary tree structure and returns a bool depending if input data was found or not.
        public bool Search(int number)
        {
            Node i = root;
            while (true)
            {
                if (i.number > number)
                {
                    if (i.left != null)
                    {
                        if (i.left.number == number)
                        {
                            return true;
                        }
                        else
                        {
                            i = i.left;
                        }
                    }
                    else 
                    {
                        break; 
                    }                    
                }
                else
                {
                    if (i.right != null)
                    {
                        if (i.right.number == number)
                        {
                            return true;
                        }
                        else
                        {
                            i = i.right;
                        }
                    }
                    else
                    {
                        break;
                    }
                    
                }
            }
            return false;
        }

        
    }

    // Node class used to create binary tree structure.
    class Node
    {
        public int number { get; set; }
        public Node left { get; set; }
        public Node right { get; set; }

        public Node(int number)
        {
            this.number = number;
        }
    }
