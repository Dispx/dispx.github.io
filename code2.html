Random Algorithms

Following is a showcase of random algorithms including: bubblesort method, string reversal method, first letter in string to uppercase method.

Code snippet

class Algorithms
    {

        // Takes a list as input and prints it out on the console.
        public void PrintList(List input)
        {
            int counter = 0;

            Console.ForegroundColor = ConsoleColor.Yellow;
            Console.WriteLine("__________");
            Console.WriteLine("Pos | Data");
            Console.WriteLine("__________");
            Console.ForegroundColor = ConsoleColor.White;

            foreach (int value in input)
            {
                Console.WriteLine(String.Format(" {0}  |  {1}", counter, value));
                counter++;
            }
        }



        // Takes a list and int value of length of the list as input and sorts it using the bubblesort.
        public void BubbleSort(List input, int n)
        {
            int i, j, temp;
            bool swapped;
            for (i = 0; i < n - 1; i++)
            {
                swapped = false;
                for (j = 0; j < n - i - 1; j++)
                {
                    if (input[j] > input[j + 1])
                    {
                        temp = input[j];
                        input[j] = input[j + 1];
                        input[j + 1] = temp;
                        swapped = true;
                    }
                }
                if (!swapped)
                {
                    break;
                }
            }
        }

        //Takes the string input and returns a string in reverse order.
        public string ReverseString(string input)
        {
            StringBuilder sb = new StringBuilder();

            for (int i = input.Length - 1; i >= 0; i--)
            {
                sb.Append(input[i]);
            }
            return sb.ToString();
        }

        //Takes string input and returns same string with first letter changed to uppercase.
        public string FirstToUpper(string input)
        {
            return input[0].ToString().ToUpper() + input.Substring(1);
        }
    }

static void Main(string[] args)

		Console.ForegroundColor = ConsoleColor.White;

		algo.PrintList(inputList);
		algo.BubbleSort(inputList, n);

		Console.ForegroundColor = ConsoleColor.Cyan;
		Console.WriteLine("-----------------------------Sorted:---------------------------------\n");
		Console.ForegroundColor = ConsoleColor.White;

		algo.PrintList(inputList);

		Console.WriteLine("Input: " + inputReverse);
